function filterTable() {
  const filter = document.getElementById('filterCriteria').value;
  const table = document.getElementById('comparisonTable');
  const rows = table.getElementsByTagName('tbody')[0].getElementsByTagName('tr');

  for (let i = 0; i < rows.length; i++) {
    const row = rows[i];
    const categories = row.getAttribute('data-category');

    if (filter === 'all' || categories.includes(filter)) {
      row.style.display = '';
    } else {
      row.style.display = 'none';
    }
  }
}

let sortAscending = true;
function sortTable() {
  const table = document.getElementById('comparisonTable');
  const tbody = table.getElementsByTagName('tbody')[0];
  const rows = Array.from(tbody.getElementsByTagName('tr'));

  rows.sort((a, b) => {
    const ratingA = a.cells[5].textContent;
    const ratingB = b.cells[5].textContent;
    return sortAscending ? ratingB.localeCompare(ratingA) : ratingA.localeCompare(ratingB);
  });

  rows.forEach(row => tbody.appendChild(row));
  sortAscending = !sortAscending;
}
</script>

  const legs = parseInt(document.getElementById('parlayLegs').value);
  const wager = parseFloat(document.getElementById('parlayWager').value);
  const odds = document.getElementById('parlayOdds').value;

  // Convert American odds to decimal

  const oddsValue = parseInt(odds);
  if (oddsValue < 0) {
    decimalOdds = (100 / Math.abs(oddsValue)) + 1;
  } else {
    decimalOdds = (oddsValue / 100) + 1;
  }

  // Calculate parlay multiplier
  const parlayMultiplier = Math.pow(decimalOdds, legs);
  const totalPayout = (wager * parlayMultiplier).toFixed(2);
  const totalProfit = (totalPayout - wager).toFixed(2);

  // Convert back to American odds for display
  let americanOdds;
  if (parlayMultiplier >= 2) {
    americanOdds = '+' + Math.round((parlayMultiplier - 1) * 100);
  } else {
    americanOdds = '-' + Math.round(100 / (parlayMultiplier - 1));
  }

  // Display results
  document.getElementById('totalPayout').textContent = totalPayout;
  document.getElementById('totalProfit').textContent = totalProfit;
  document.getElementById('parlayOddsDisplay').textContent = americanOdds;

}
