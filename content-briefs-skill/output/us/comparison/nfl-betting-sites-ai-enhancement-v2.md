<script>
// ==========================================

document.addEventListener('DOMContentLoaded', function() {

  // Initialize tabbed review system
  initReviewTabs();

  console.log('📊 Review tabs system ready');
});

// ==========================================
// TABBED REVIEW SYSTEM
// ==========================================
function initReviewTabs() {
  const tabs = document.querySelectorAll('.review-tab');
  const contents = document.querySelectorAll('.review-tab-content');

  tabs.forEach(tab => {
    tab.addEventListener('click', function() {
      const targetId = this.getAttribute('data-tab');

      // Remove active from all
      tabs.forEach(t => t.classList.remove('active'));
      contents.forEach(c => c.classList.remove('active'));

      // Activate clicked
      this.classList.add('active');
      document.getElementById(targetId).classList.add('active');

      console.log('📑 Tab switched to: ' + targetId);
    });
  });
}

// ==========================================
// UNIQUE FEATURES TOGGLES
// ==========================================
function toggleUniqueFeatures(brandId) {
  const content = document.getElementById('features-' + brandId);
  const icon = document.getElementById('toggle-icon-' + brandId);

  if (content.classList.contains('active')) {
    content.classList.remove('active');
    icon.textContent = '▶';
  } else {
    content.classList.add('active');
    icon.textContent = '▼';
  }

  console.log('🔄 Feature toggle for: ' + brandId);
}
</script>
