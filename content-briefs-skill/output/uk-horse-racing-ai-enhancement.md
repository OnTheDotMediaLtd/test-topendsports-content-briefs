  const stake = parseFloat(document.getElementById('ewStake').value);
  const winOdds = parseFloat(document.getElementById('ewWinOdds').value);
  const placeTerms = parseFloat(document.getElementById('ewPlaceTerms').value);

  // Each-way = 2 × stake
  const totalStake = stake * 2;

  // Win calculation
  const winReturn = (stake * winOdds) + stake;

  // Place calculation
  const placeOdds = 1 + placeTerms;
  const placeReturn = (stake * placeOdds) + stake;

  // If only place wins (not win)
  const onlyPlace = placeReturn - stake;

  document.getElementById('ewTotalStake').textContent = totalStake.toFixed(2);

  console.log('✓ Each-way calculator executed: £' + stake + ' at ' + winOdds + ' odds');
}
