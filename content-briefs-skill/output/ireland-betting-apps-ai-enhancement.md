### Sortable Comparison Table JavaScript
```javascript
function initComparisonTableSort() {
  const table = document.querySelector('.betting-apps-comparison');
  const headers = table.querySelectorAll('thead th');

  headers.forEach((header, index) => {
    header.style.cursor = 'pointer';
    header.addEventListener('click', () => sortTable(table, index));
  });
}

function sortTable(table, columnIndex) {
  const tbody = table.querySelector('tbody');
  const rows = Array.from(tbody.querySelectorAll('tr'));

  rows.sort((a, b) => {
    const aValue = a.cells[columnIndex].textContent.trim();
    const bValue = b.cells[columnIndex].textContent.trim();

    // Numeric sorting for ratings
    if (columnIndex === 1 || columnIndex === 2) {
      return parseFloat(bValue) - parseFloat(aValue);
    }

    return aValue.localeCompare(bValue);
  });

  rows.forEach(row => tbody.appendChild(row));
}

document.addEventListener('DOMContentLoaded', initComparisonTableSort);
```
