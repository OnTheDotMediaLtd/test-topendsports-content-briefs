// Sort table functionality
let sortDirection = {};

function sortTable(columnIndex) {
    const table = document.getElementById('ukBettingTable');
    const tbody = table.querySelector('tbody');
    const rows = Array.from(tbody.querySelectorAll('tr'));

    // Toggle sort direction
    if (!sortDirection[columnIndex]) sortDirection[columnIndex] = 'asc';
    else sortDirection[columnIndex] = sortDirection[columnIndex] === 'asc' ? 'desc' : 'asc';

    rows.sort((a, b) => {
        let aValue, bValue;

        if (columnIndex === 3) { // Min Deposit - numeric
            aValue = parseFloat(a.cells[columnIndex].getAttribute('data-value'));
            bValue = parseFloat(b.cells[columnIndex].getAttribute('data-value'));
        } else if (columnIndex === 4) { // Rating - numeric
            aValue = parseFloat(a.cells[columnIndex].getAttribute('data-value'));
            bValue = parseFloat(b.cells[columnIndex].getAttribute('data-value'));
        } else {
            aValue = a.cells[columnIndex].textContent.trim();
            bValue = b.cells[columnIndex].textContent.trim();
        }

        if (sortDirection[columnIndex] === 'asc') {
            return aValue > bValue ? 1 : -1;
        } else {
            return aValue < bValue ? 1 : -1;
        }
    });

    rows.forEach(row => tbody.appendChild(row));
    console.log('✅ Table sorted by column ' + columnIndex + ' in ' + sortDirection[columnIndex] + ' order');
}

// Filter by sport
function filterBySport() {
    const sportFilter = document.getElementById('sportFilter').value;
    const rows = document.querySelectorAll('#ukBettingTable tbody tr');

    rows.forEach(row => {
        if (sportFilter === 'all') {
            row.style.display = '';
        } else {
            const sports = row.getAttribute('data-sports');
            if (sports && sports.includes(sportFilter)) {
                row.style.display = '';
            } else {
                row.style.display = 'none';
            }
        }
    });

    console.log('✅ Filtered by sport: ' + sportFilter);
}

// Reset all filters
function resetFilters() {
    document.getElementById('sportFilter').value = 'all';
    const rows = document.querySelectorAll('#ukBettingTable tbody tr');
    rows.forEach(row => row.style.display = '');
    console.log('✅ Filters reset');
}
</script>

let sortDirection = {};

function sortTable(columnIndex) {
    const table = document.getElementById('ukBettingTable');
    if (!table) return;

    const tbody = table.querySelector('tbody');
    const rows = Array.from(tbody.querySelectorAll('tr'));

    if (!sortDirection[columnIndex]) sortDirection[columnIndex] = 'asc';
    else sortDirection[columnIndex] = sortDirection[columnIndex] === 'asc' ? 'desc' : 'asc';

    rows.sort((a, b) => {
        let aValue, bValue;

        if (columnIndex === 3 || columnIndex === 4) {
            aValue = parseFloat(a.cells[columnIndex].getAttribute('data-value'));
            bValue = parseFloat(b.cells[columnIndex].getAttribute('data-value'));
        } else {
            aValue = a.cells[columnIndex].textContent.trim();
            bValue = b.cells[columnIndex].textContent.trim();
        }

        if (sortDirection[columnIndex] === 'asc') {
            return aValue > bValue ? 1 : -1;
        } else {
            return aValue < bValue ? 1 : -1;
        }
    });

    rows.forEach(row => tbody.appendChild(row));
    console.log('✅ Table sorted by column ' + columnIndex + ' in ' + sortDirection[columnIndex] + ' order');
}
