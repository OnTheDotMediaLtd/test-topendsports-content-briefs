<script>
function sortTable(n) {
  const table = document.querySelector('.free-bets-comparison');
  let switching = true;
  let dir = "asc";
  let switchcount = 0;

  while (switching) {
    switching = false;
    const rows = table.getElementsByTagName('tr');
    for (let i = 1; i < rows.length - 1; i++) {
      let shouldSwitch = false;
      const x = rows[i].getElementsByTagName('td')[n];
      const y = rows[i + 1].getElementsByTagName('td')[n];

      if (n === 1 || n === 3 || n === 4) {
        const xVal = parseInt(x.textContent.replace(/[€x]/g, ''));
        const yVal = parseInt(y.textContent.replace(/[€x]/g, ''));
        if (dir === "asc" && xVal > yVal) shouldSwitch = true;
        if (dir === "desc" && xVal < yVal) shouldSwitch = true;
      } else {
        if (dir === "asc" && x.textContent > y.textContent) shouldSwitch = true;
        if (dir === "desc" && x.textContent < y.textContent) shouldSwitch = true;
      }

      if (shouldSwitch) {
        rows[i].parentNode.insertBefore(rows[i + 1], rows[i]);
        switching = true;
        switchcount++;
        break;
      }
    }
    if (switchcount === 0 && switching) dir = dir === "asc" ? "desc" : "asc";
  }
}
</script>

function calculateValue() {
  const freeBet = parseFloat(document.getElementById('freeBetAmount').value);
  const odds = parseFloat(document.getElementById('odds').value);
  const wagering = parseInt(document.getElementById('wagering').value);
  const minOdds = parseFloat(document.getElementById('minOdds').value);

  if (!freeBet || !odds || !wagering || !minOdds) return alert('Please fill all fields');
  if (odds < minOdds) return alert(`Odds must be at least ${minOdds}`);

  const profitIfWin = freeBet * (odds - 1);
  const expectedValue = profitIfWin * (1 / wagering);

  document.getElementById('expectedValue').textContent = expectedValue.toFixed(2);
  document.getElementById('result').style.display = 'block';
}

document.addEventListener('DOMContentLoaded', function() {
  calculateValue();
});
