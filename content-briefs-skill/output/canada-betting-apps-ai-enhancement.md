document.addEventListener('DOMContentLoaded', function() {
  const appSearch = document.getElementById('appSearch');
  const ratingFilter = document.getElementById('ratingFilter');
  const resetBtn = document.getElementById('resetBtn');
  const table = document.getElementById('comparisonTable');

  function filterTable() {
    const searchTerm = appSearch.value.toLowerCase();
    const minRating = parseFloat(ratingFilter.value) || 0;
    const rows = table.querySelectorAll('tbody tr');

    rows.forEach(row => {
      const brandName = row.querySelector('.brand-name').textContent.toLowerCase();
      const iosRating = parseFloat(row.querySelector('.ios-rating .score')?.textContent) || 0;

      const matchesSearch = brandName.includes(searchTerm);
      const matchesRating = iosRating >= minRating;

      row.style.display = matchesSearch && matchesRating ? '' : 'none';
    });
  }

  appSearch.addEventListener('input', filterTable);
  ratingFilter.addEventListener('change', filterTable);
  resetBtn.addEventListener('click', function() {
    appSearch.value = '';
    ratingFilter.value = '';
    filterTable();
  });
});
</script>
